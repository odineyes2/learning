# module

## commonjs

```javascript | var.js
const odd = "odd";
const even = 'even';
module.exports = {
    odd, even,
};
```

```javascript | function.js
const {odd,even} = require('./var.js');
const checkFn = (num)=>{
    if(num%2){
        return odd
    } return even;
module.exports = checkFn;
}
```
